var thing={
	    addHandler:function(ele,str,fun){
  ele.addEventListener ? ele.addEventListener(str,fun,false):ele.attachEvent("on"+str,fun)
	},
	    removeHandler:function(ele,str,fun){
	ele.removeEventListener ? ele.removeEventListener(str,fun,false):ele.detachEvent("on"+str,fun)
	},
	    addLoadHandler: function (fnHandler) {
		this.addHandler(window, "load", fnHandler);
	}
};
